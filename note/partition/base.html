# Pow

实现Pow(x, n) 即x^n x的n次幂函数

方法1：暴力解法 O(n)

result = 1
for i=0;i<len-1 {
    result *=x
}

方法2：分治 

1. 终结条件
2. 处理当前层逻辑
3. 调用这个函数下探一层
4. 合并结果并返回

x^n ------>  2^10 :2^5*2^5---->(2^2*2^2)*2   奇数再补一个x   
            2^10 ---> 2^5 --->  (2^2)*2
pow(x, n):
    求子问题 subproble: subresult = pow(x, n/2) ---->时间复杂度log(N)
merge:  如果n是奇数再补一个x
    if n%2 ==1 { // 奇数
        result = subresult*subresult*x
    }else{ // 偶数
        result = subresult*subresult
    }

# 子集 78

var subsets = function (nums) {
    const result = [];
    dfs(nums, result, [], 0);
    return result;
};

dfs = (nums, result, array, index) => {
    // 走到最末层了
    if (index === nums.length) {
        result.push([...array]);
        return;
    }

    // 当前这个元素有两种选择：选 or 不选
    // 不选，array不需要操作
    dfs(nums, result, array, index + 1);

    // 选，把当前的index值加进去
    array.push(nums[index]);
    dfs(nums, result, array, index + 1);

    // reverse the current state 把最后加的数从里面去掉
    //（回溯：需要清除状态）
    array.pop();
}

console.log(subsets([1,2,3]))

var subsets = function(nums) {
    // return nums.reduce((total,num)=>total.concat(total.map(item=>item.concat(num))),[[]])
    return nums.reduce((total,num)=>{
      return  total.concat(total.map(item=>{
          return item.concat(num)
      }))
    },[[]])
};
