动态规划 = 递归+记忆化 动态规划解决方案从底部开始解决问题，将所有小问题解决掉，然后合并成一个整体解决方案，从而解决掉整个大问题。 递归是从顶部开始将问题分解，通过解决掉所有分解出小问题的方式，来解决整个问题。

不要额外使用空间

// 动态规划：求最长最短、最多最少 用max min；是否要累加，要的话max/min后再加
1. 重复性（分治） problem(i, j) = min(sub(i+1, j), sub(i+1, j+1)) +  a[i, j] // 分治，找重复的； 如果要累加max\min后再家
2. 定义状态数组 f[i, j]   // 确定状态是一维数组还是二维，在这个位置的值代表什么
3. dp方程 f[i, j] = min(f[i+1, j], f[i+1, j+1])+a[i,j]    // f[i, j] 表示在[i, j]点可以求的最大/小或可走的数

dp方程：F(n) = F(n-1) + F(n-2) // F表示上到第n级台阶所可以走的步数

自顶向下怎么dp，自底向上怎么dp

动态规划关键点：
1. 最优子结构 opt[n] = best_of(opt[n-1], opt[n-2], ...)
2. 存储中间状态： opt[i]
3. 递推公式：状态转义方程（dp方程
Fib: opt[i] = opt[n-1]+opt[n-2]  （考虑memo[n]缓存）

二维路径：opt[i, j] = opt[i+1][j] + opt[i][j+1]（且判断a[i, j]是否空地）

// 路径计数求解
opt[i, j] = opt[i+1, j] + opt[i, j+1]
if a[i, j] ='空地'
  opt[i, j] =opt[i+1, j]+opt[i, j+1]

const fib = function(n) {
  if(n<=1) return n // 把0和1两种状态都处理了
  const cache = []
  cache[0]=0
  cache[1]=1

  /* 检查创建的cacahe里面有没有momoize里要找的数，
      如果有就直接return里面的值，如果没有就再用f(n-1) + f(n-2)
  */
  function momeize(num){
      if(cache[num]!==undefined){
          return cache[num]
      }
      cache[num] =( momeize(num-1)+momeize(num-2)) % (1e9 +7)
      return cache[num] 
  }

  return momeize(n)
}

const fib = function(n) {
    if(n<=1) return n 
    
    let prev2 = 0 // 倒数第二个数
    let prev1 = 1 // 倒数第一个数
    let result = 0  
    for(let i=2;i<=n;i++){
        // result等于倒数第一个数+倒数第二个数
        result = prev1 + prev2
        // 让prev1、prev2都往前进一个值
        prev2 = prev1 // 旧的prev2等于新的prev1
        prev1 = result // 旧的prev1等于新的result， 即下一个result=上一个result+prve1

    }
    return result
}

# 不同路径 II

一个机器人位于一个 m x n 网格的左上角 （起始点在下图中标记为“Start” ）。
机器人每次只能向下或者向右移动一步。机器人试图达到网格的右下角（在下图中标记为“Finish”）。
现在考虑网格中有障碍物。那么从左上角到右下角将会有多少条不同的路径？
网格中的障碍物和空位置分别用 1 和 0 来表示。
思路：只能向右/下移，想到 dp[当前] = dp[上] + dp[左]，考虑几种边界情况需要注意 

function uniquePathsWithObstacles(arr, m, n){
  let dp=(m, n)=>{
      // 检查起始或者目标元素是不是1（被占用了），如果起始或者最后那个格就是1，说明怎么都怎么不到那，直接返回0
      if (arr[m - 1][n - 1] === 1 || arr[0][0] === 1) {
        return 0
      }
      if(m===2&&n===2){
        // 如果是情况1：0此路不通; 情况2：左有障碍物或者右边有障碍物，那么可能有一个不没有障碍物，所以为1，其他情况 都能走通为2
        return (arr[1][1]===1 || arr[1][0]+arr[0][1]===2)?0:(arr[1][0]===1||arr[0][1]===1)?1:2
      }else if(m<2 || n<2){
        // 单行情况，有障碍物，不通为0，通为1
        if(m<2){
          return arr[m - 1].includes(1) ? 0 : 1
        }else{
          // 单列有障碍物为0，没有为1
          for(let i=0; i<m; i++){
            if(arr[i][0]===1){
              return 0
            }
          }
          return 1
        }
      }else{
        // 不是上面的情况（不是边界，状态为方程）
        return dp(m-1, n) + dp(m, n-1)
      }
  }
  return dp(m, n)
}
let arr = [
  [0,0,0],
  [0,1,0],
  [0,0,0]
]
console.log(uniquePathsWithObstacles(arr, 3, 3))  // 2

# 爬楼梯

假设你正在爬楼梯。需要 n 阶你才能到达楼顶。
每次你可以爬 1 或 2 个台阶。你有多少种不同的方法可以爬到楼顶呢？ 动态规划：需要记忆化的数组

台阶怎么走：递推
1: 1 阶 + 1 阶
2: 2 阶
3: 1 阶 + 1 阶 + 1 阶 || 1 阶 + 2 阶 || 2 阶 + 1 阶
3:=> f(1) + f(2) 最后走两步或者一步
4:=> f(2) + f(3)

斐波那契数列以如下被以递推的方法定义：F(1)=1，F(2)=1, F(n)=F(n - 1)+F(n - 2)

技巧：找 最近 重复子问题
重复可用if else 、for while recursion

//时间复杂度为 O(n)
function climbStairs(n){ 
    const memo = []
    memo[1] = 1
    memo[2] = 2
    for(let i=3;i<=n;i++){  
        memo[i] = memo[i-2] + memo[i-1]
    }
    return memo[n]
}
console.log(climbStairs(3))

# 最长公共子序列

输入：text1 = "abcde", text2 = "ace"
输出：3
解释：最长公共子序列是 "ace"，它的长度为 3

顺序不能变

字符串进行变化的dp，会把它扩展成为一个二维数组来定义状态

  • 方法1：暴力解法，枚举text1，对每个字母都是取和不取，生成一个子序列，然后看这个子序列，是否也在text2存在
  • 方法2：找重复性，化解为febonacci
1. S1=''    // 只要有一个空，就为空
   S2=任意字符串
  
2. S1='A'    // A是否存在任意字符串中
   S2=任意

// 最后字母都是相同的，从最后字符开始看起，因为A和A是相同的，就可以转化为求S1和S2前面的字符串的最长公共子序列
3. S1='....A'   
   S2='...A'

状态转移方程:
两字符串相同: dp[i][j] = dp[i-1][j-1] + 1;
不同: dp[i][j] = Math.max(dp[i-1][j],dp[i][j-1])

var longestCommonSubsequence = function(text1, text2) {
   let m = text1.length;
   let n = text2.length;
   let dp = Array.from({length: m+1},()=> new Array(n+1).fill(0));
   dp[0][0] = 0;
   for(let i = 1; i <=m; i++) {
       for(let j = 1; j <=n; j++) {
           if(text1[i-1] === text2[j-1]) {
               dp[i][j] = dp[i-1][j-1] + 1;
           } else {
               dp[i][j] = Math.max(dp[i-1][j],dp[i][j-1]);
           }
       }
   }
   return dp[m][n];
};
console.log(longestCommonSubsequence("abcde", "ace" ))

# 三角形最小路径和

类似左括号右括号 或 找零钱 或 爬楼梯（每一步往左走，往右走） 问题

  • 方法1：递归，n层递归，可以左或者右
  • 方法2：动态规划（至下而上累加）

拆分出来就是：求3和4到下面的最小路径是多少，从3和4的最小路径和里面找一个较小者，再加上自己的2，就变成了2走到最下面的最小路径和 2+3 和 2+4 比较，返回最小的结果再加2

1. 重复性（分治） problem(i, j) = min(sub(i+1, j), sub(i+1, j+1)) + a[i, j]  // a[i, j] 为自己本身
2. 定义状态数组 f[i, j]
3. dp方程 f[i, j] = min(f[i+1, j], f[i+1, j+1])+a[i, j]

// dp不是0的二维数组，而是里面有值的二维数组，好处是不用再加上i,j
// 如果不考虑副作用，直接在原数组triangle上修改
var minimumTotal = function(triangle) {
    let dp=triangle
    for(let i = dp.length-2;i>=0 ;i--){
        for(let j=0; j< dp[i].length; j++){
            dp[i][j] = Math.min(dp[i+1][j], dp[i+1][j+1]) + dp[i][j];
        }
    }
    return dp[0][0]
};

# 最大子序和

之前都是负就丢掉，只求i-1最大的值 + 之前累加的值

dp： 从后往前
1. 分治（子问题） max_sum(i) = Max(max_sum(i-1), 0) + a[i]n   // 假设不包含i-1为0
2. 状态数组定义   f[i]  // 表示从0到i，且第i个元素被累加进来的话的最大的sum
3. dp方程(状态转义方程)：f[i] = Math(f[i-1], 0) + a[i]  // 扫描这个数组，每一个f[i]就等于Math(f[i-1], 0) + a[i]，找出最大的f[i]
// 不断用sum进行累加，累加到负的就丢掉，不然就继续保留着，直到每一步都找到最大值

/*
  dp公式：dp[i]=max[nums[i], nums[i]+dp[i-1]]
  最大子序和 = 当前元素自身最大 或者 包含之前后最大
*/
var maxSubArray = function(nums) {
    if(nums == null) return 0
    let dp = []
    dp[0] = nums[0]
    let max = dp[0]
    for(let i=1; i<nums.length; i++){
        // nums[i-1]代表dp[i-1]
        // dp[i] = Math.max(nums[i], nums[i]+dp[i-1])
        dp[i] = Math.max(0, dp[i-1]) + nums[i]
        max = Math.max(dp[i], max)
    }
    return max
};
console.log(maxSubArray([-2,1,-3,4,-1,2,1,-5,4]))

类似爬楼梯

把总价格减下一去

subproblems:
dp array：f(n) = min(f(n-k)(k in [1, 2, 5]))  // f(n)=f(n-1)+f(n-2)+f(n-3) 可以走1步，可以2步，想象coins是走多少步，但是这里是数组，所以n-k，然后从所有的n-k中找最小值

const coinChange = (coins, amount) => {
    let dp = new Array(amount + 1).fill(Infinity) //最大的值
    dp[0] = 0  // 上第0级台阶，走0步
    // 把之前的步数遍历一次，每次都coins[j]步  步数=取多少面值
    for(let i=1; i<=amount; i++){
        for(let j=0;j<coins.length; j++){
            // i表示要凑的面值，如果面积i剩余1，找出面值2，那就超了
            if(coins[j]<=i){
                dp[i]=Math.min(dp[i], dp[i-coins[j]]+1) // i要凑的面值- coins[j]找出来的钱 = 剩余的面值
            }
        }
    }
   
    return dp[amount] > amount ? -1 : dp[amount]
}

# 打家劫舍

/*
求：从0到第i个房子能偷到最大的值

第i个房子的递推 = 从零到i-1间的最大值（可以偷了i-1天的房子，可能没偷，如果偷就不能偷后面房子即不能加后面的数） + 偷第i间
a[i] = a[i-1] + nums[i]

第i个房子有没有偷，所以再开多一维记录
a[i][0, 1]:0表示第i个房子不偷，1表示偷

从0到i个房子能偷的最大值，同时第i个房子不偷的话
a[i][0] = Max(a[i-1][0], a[i-1][1])  // 0到i-1前面的房子可以偷，也可以不偷 
a[i][1] = Max(a[i-1][0], 0) + nums[i] = a[i-1][0] + nums[i]  // 要偷第i天房子，那么i-1的就不能偷
*/

// 时间复杂度O(N)
var rob = function (nums) {
  if(nums == null || nums.length == 0) return 0
  let len = nums.length
  let a = []
  for (var i = 0; i < len; i++) {
      a[i] = new Array();
  }

  a[0][0] = 0
  a[0][1] = nums[0]

  for (let i = 1; i < len; ++i) {
      a[i][0] = Math.max(a[i - 1][0], a[i - 1][1])
      a[i][1] = a[i - 1][0] + nums[i]
  }

  return Math.max(a[len - 1][0], a[len - 1][1])
};

console.log(rob([1, 2, 3, 1]))

优化
a[i]：0到i间，且nums[i]必偷的最大值
// a[i] = Max(a[i-1], a[i-2]+nums[i], a[i-3]+nums[i]...)
a[i] = Max(a[i-1], a[i-2]+nums[i])
偷第i-1间的最大值，且i-1间偷了，就不能偷第i间
偷第i-2间的最大值，且i-2间偷了，那我就能偷第i间

var rob = function (nums) {
  if (nums == null || nums.length == 0) return 0
  if (nums ==1) return nums[0]
  let len = nums.length
  let a = []

  a[0] = nums[0]
  a[1] = Math.max(nums[0], nums[1])

  let res = Math.max(a[0], a[1])

  for (let i = 2; i < len; ++i) {
      a[i] = Math.max(a[i-1], a[i-2]+nums[i])
      res = Math.max(res, a[i])
  }

  return res
};

console.log(rob([1, 2, 3, 1]))

# 剪绳子

1. 重复性（分治） f(n)=f(n-i)*f(i)
2. 定义状态数组 f(n)  
3. dp方程 f(n) = Math.max(f(n), f(i)*f(n-i))

f(i) = f(i-j) f(j)
f(n-i) = f(n-i-j) f(j)

var cuttingRope = function(n) {
  let dp = Array.from({length: n+1},()=>1);
  if (n == 2) {
      return 1;
  }
  if (n == 3) {
      return 2; // 1 + 2
  }
  dp[1] = 1;
  dp[2] = 2;
  dp[3] = 3; // 1 + 2
  for (let k = 1; k <= n; k++) {
      for (let i = 1; i <= k / 2; i++) {
          dp[k] = Math.max(dp[k], dp[i] * dp[k - i]);
      }
  }
  return dp[n];
};
console.log(cuttingRope(2))

var cuttingRope = function(n) {
  let dp = Array.from({length: n+1},()=>1);
  if (n == 2) {
      return 1;
  }
  if (n == 3) {
      return 2; // 1 + 2
  }
  dp[1] = BigInt(1) ;
  dp[2] = BigInt(2);
  dp[3] = BigInt(3) ; // 1 + 2
  for (let k = 1; k <= n; k++) {
      for (let i = 1; i <= k / 2; i++) {
          let temp = dp[i] * dp[k-i];
          if(dp[k]<temp){
              dp[k] = temp
          }
      }
  }
  return parseFloat((dp[n] % BigInt(1000000007)).toString());
};

    for(let i = 0; i < xLen + 2; i ++) {//最外层都是0，就不用考虑边界问题了，下面的状态转移方程也就优一点变化，但中心思想不变。
        let arr = new Array(yLen + 2).fill(0);
        dp[i] = arr;
    }*/
    let dp = Array.from({length:xLen+1}, () => new Array(yLen+1).fill(0)); // 注意边界
    for(let x = 0; x < xLen; x ++) {
        for(let y = 0; y < yLen; y ++) {
            dp[x + 1][y + 1] = Math.max(dp[x][y + 1], dp[x + 1][y]) + grid[x][y];
        }
    }
    return dp[xLen][yLen];
};
