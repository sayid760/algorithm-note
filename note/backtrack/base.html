# 电话号码的字母组合

方法：递归 1）把电话号码的字母转成map 2）通过递归获取到'abc' 'def'，遍历'def'，拿到每个值跟之前的str('a','b','c')拼接成'ad'...

function letterCombinations(digits){
    const result = []
    const map = new Map()
    map.set('2','abc');
    map.set('3','def');
    map.set('4','ghi');
    map.set('5','jkl');
    map.set('6','mno');
    map.set('7','pqrs');
    map.set('8','tuv');
    map.set('9','wxyz');
    combinations('', digits, result, 0, map);
    return result
}

// 输出(变)，输入，最终返回数据，level(变), 其他辅助数据
function combinations(str, digits, result, index, map){ 
    if(index == digits.length){
        result.push(str)
        return 
    }
    let temp = map.get(digits[index])  'abc'  'def'
    for(let i=0; i<temp.length; i++){
        combinations(str+temp[i], digits, result, index+1, map)
    }
}

方法2：分割数字，找到对应的字母，拿到之后递归做两两运算（两层for循环获取各自的字母push到result里） 数组从0删除两项， 

const letterCombinations = (str) => {
  // 对输入做处理，如果小于1返回空（LeetCode测试用例）
  if (str.length < 1) return []
  // 建立电话号码键盘映射
  let map = ['', 1, 'abc', 'def', 'ghi', 'jkl', 'mno', 'pqrs', 'tuv', 'wxyz']
  // 如果只给了一个按键，直接把按键内容取出来并按单个字符分组就可以了（LeetCode测试用例）
  if (str.length < 2) return map[str].split('')
  // 把输入字符串按单字符分隔变成数组，234=>[2,3,4]
  let num = str.split('')
  // 保存键盘映射后的字母内容，如 23=>['abc','def']
  let code = []
  num.forEach(item => {
    if (map[item]) {
      code.push(map[item])
    }
  })
  let comb = (arr) => {
    // 临时变量用来保存前两个组合的结果
    let tmp = []
    // 最外层的循环是遍历第一个元素，里层的循环是遍历第二个元素
    for (let i = 0, il = arr[0].length; i < il; i++) {
      for (let j = 0, jl = arr[1].length; j < jl; j++) {
        tmp.push(`${arr[0][i]}${arr[1][j]}`)
      }
    }
    arr.splice(0, 2, tmp) // [["ad", "ae", "af", "bd", "be", "bf", "cd", "ce", "cf"]]
    if (arr.length > 1) {
      comb(arr)
    } else {
      return tmp
    }
    return arr[0]
  }
  return comb(code)
}

console.log(letterCombinations("23"))

# N皇后

竖、左对角线、右对象线 如果存在就有攻击，相等于下一层的n，n-1，n+1位置都有攻击

1）判断条件 列撇捺 设置为Set，用来判断有没有占用了列撇捺
2）调递归dfs，逻辑：在当前这一层遍历所有列，遍历列的时候把皇后放到列上面，然后更新它所占位的撇捺还有列，然后再drill down到下一层，再reverse state状态恢复，把当前层已经改变的状态去掉
3）输出整个棋盘result

var solveNQueens = function (n) {
  var obj = {}
  var pie = [] // 撇
  var na = []  // 捺
  var res = [] // 最终返回结果
  trackback(0)
  function trackback(row) {
      // 1.满足条件（横数==4）终结
      if (row == n) {
          var arr = []
          for (var i in obj)
          /* 
              '1'.padStart(3,'0') ---> '001'
              '1'.padStart(3,'0').padEnd(4,'0')  --->  "0010"

              obj ->  {0: 1, 1: 3, 2: 0, 3: 2} 
            *  [".Q..",  // 解法 1
              "...Q",
              "Q...",
              "..Q."],
            */
              arr.push('Q'.padStart(obj[i] + 1, '.').padEnd(n, '.')) 
          res.push(arr)
          return
      }

      // 2. 处理当前层逻辑
      // 遍历四层中，记录每一层中不会被攻击的位置
      for (var i = 0; i < n; i++) {
          if (Object.values(obj).indexOf(i) != -1 || na.indexOf(row + i) != -1 || pie.indexOf(row - i) != -1)
              continue
          obj[row] = i   // 记录第几层的哪个位置没被攻击 
          pie.push(row - i)  // 记录下来，用于下一轮循环判断
          na.push(row + i) //  [0, 4, 3]
          // 3. 下探一层
          trackback(row + 1) 
          // 4. reverse state 恢复当前层的状态
          obj[row] = -1   
          na.pop()   // 用完就清空  
          pie.pop()
      }
  }
  return res
};
console.log(solveNQueens(4))
